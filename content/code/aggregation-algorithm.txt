FUNCTION CreateAggregatedSummary(U)
  INPUT:
    list of sources U
  OUTPUT:
    combined summary for each quad component: S.subject, S.predicate, S.object S.graph
FOREACH c in [subject, predicate, object, graph]
  S.c = Summary_Initialize()
FOREACH u in U
  S' = get summaries from u
  FOREACH c in [subject, predicate, object, graph]
    S.c = Summary_Combine(S.c, S'.c)
RETURN S
