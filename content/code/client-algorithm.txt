FUNCTION SelectSources(q, K, Σ, U)
  INPUT:
    q: quad pattern query
    K: access keys
    Σ: summary containing: Σ.subject, Σ.predicate, Σ.object, Σ.graph
    U: set of sources
  OUTPUT:
    U': list of selected sources
U' = []
FOREACH c in [subject, predicate, object, graph]
  IF q.c not variable
    FOREACH k in K
      IF ! SummaryContains(Σ.c, q.c, k, ε)
        RETURN []
      FOREACH u in U
        IF ! SummaryContains(Σ.c, q.c, k, u)
          add u to U'
RETURN U'
