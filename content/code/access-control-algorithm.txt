FUNCTION ProcessRequest(c, q, QPK)
  INPUT:
    client credentials c
    query q
    QPK a hashmap relating quads to policies and keys
  OUTPUT:
    query results R
R = {}
FOREACH p in GetPolicies(QPK, c, q)
  IF p.type = prohibition & cr = prohibition_overrides
    break
  ELSE
    R = ExecuteQueryWithAccessControl(c, q, p)
    break
RETURN R
